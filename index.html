CPU Scheduling Programs

1. FCFS (First Come First Serve)


#include <stdio.h>

int main() {
    int n;
    int AT[10], BT[10], WT[10], TT[10], CT[10];
    float Avg_WT, Avg_TT;
    int Total_WT = 0, Total_TT = 0;

    printf("Enter the number of processes (max 10): ");
    scanf("%d", &n);

    printf("Enter the Arrival time and Burst time for each process:\n");
    for (int i = 0; i < n; i++) {
        printf("Process P%d:\n", i + 1);
        printf("  Arrival Time (AT): ");
        scanf("%d", &AT[i]);
        printf("  Burst Time (BT): ");
        scanf("%d", &BT[i]);
    }

    int current_time = 0;
    for (int i = 0; i < n; i++) {
        if (current_time < AT[i]) {
            current_time = AT[i];
        }
        WT[i] = current_time - AT[i];
        Total_WT += WT[i];
        current_time += BT[i];
        CT[i] = current_time;
        TT[i] = CT[i] - AT[i];
        Total_TT += TT[i];
    }

    Avg_WT = (float)Total_WT / n;
    Avg_TT = (float)Total_TT / n;

    printf("\nProcess\tArrival\tBurst\tWaiting\tTurnaround\n");
    printf("--------------------------------------------\n");
    for (int i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t%d\t%d\n", i+1, AT[i], BT[i], WT[i], TT[i]);
    }

    printf("\nAverage Waiting Time: %.2f\n", Avg_WT);
    printf("Average Turnaround Time: %.2f\n", Avg_TT);
    return 0;
}

2. SJF (Shortest Job First)


#include <stdio.h>

int main() {
    int n, temp, AT[10], BT[10], WT[10], TT[10], P[10];
    int Total_WT = 0, Total_TT = 0;
    float Avg_WT, Avg_TT;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    printf("Enter Arrival Time and Burst Time:\n");
    for (int i = 0; i < n; i++) {
        printf("P%d Arrival: ", i+1);
        scanf("%d", &AT[i]);
        printf("P%d Burst: ", i+1);
        scanf("%d", &BT[i]);
        P[i] = i+1;
    }

    for (int i = 0; i < n-1; i++) {
        for (int j = i+1; j < n; j++) {
            if (BT[i] > BT[j]) {
                temp = BT[i]; BT[i] = BT[j]; BT[j] = temp;
                temp = AT[i]; AT[i] = AT[j]; AT[j] = temp;
                temp = P[i]; P[i] = P[j]; P[j] = temp;
            }
        }
    }

    WT[0] = 0;
    for (int i = 1; i < n; i++) {
        WT[i] = 0;
        for (int j = 0; j < i; j++)
            WT[i] += BT[j];
        Total_WT += WT[i];
    }

    printf("\nProcess\tArrival\tBurst\tWaiting\tTurnaround\n");
    printf("--------------------------------------------\n");
    for (int i = 0; i < n; i++) {
        TT[i] = BT[i] + WT[i];
        Total_TT += TT[i];
        printf("P%d\t%d\t%d\t%d\t%d\n", P[i], AT[i], BT[i], WT[i], TT[i]);
    }

    Avg_WT = (float)Total_WT/n;
    Avg_TT = (float)Total_TT/n;

    printf("\nAverage Waiting Time: %.2f", Avg_WT);
    printf("\nAverage Turnaround Time: %.2f", Avg_TT);
    return 0;
}

3. SRTF (Shortest Remaining Time First)


#include <stdio.h>
#include <limits.h>

int main() {
    int n, AT[10], BT[10], RT[10], WT[10], TT[10], CT[10];
    int complete = 0, t = 0, minm = INT_MAX, shortest = 0, finish_time;
    int check = 0;
    float total_WT = 0, total_TT = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        printf("P%d Arrival: ", i+1);
        scanf("%d", &AT[i]);
        printf("P%d Burst: ", i+1);
        scanf("%d", &BT[i]);
        RT[i] = BT[i];
    }

    while (complete != n) {
        for (int j = 0; j < n; j++) {
            if ((AT[j] <= t) && (RT[j] < minm) && RT[j] > 0) {
                minm = RT[j];
                shortest = j;
                check = 1;
            }
        }
        if (check == 0) { t++; continue; }

        RT[shortest]--;
        minm = RT[shortest];
        if (minm == 0) minm = INT_MAX;

        if (RT[shortest] == 0) {
            complete++;
            check = 0;
            finish_time = t + 1;
            WT[shortest] = finish_time - BT[shortest] - AT[shortest];
            if (WT[shortest] < 0) WT[shortest] = 0;
            TT[shortest] = finish_time - AT[shortest];
            total_WT += WT[shortest];
            total_TT += TT[shortest];
        }
        t++;
    }

    printf("\nProcess\tArrival\tBurst\tWaiting\tTurnaround\n");
    printf("--------------------------------------------\n");
    for (int i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t%d\t%d\n", i+1, AT[i], BT[i], WT[i], TT[i]);
    }

    printf("\nAverage Waiting Time: %.2f\n", total_WT/n);
    printf("Average Turnaround Time: %.2f\n", total_TT/n);
    return 0;
}

4. Priority Scheduling


#include <stdio.h>

int main() {
    int n, P[10], AT[10], BT[10], PR[10], WT[10], TT[10];
    int i, j, pos, temp, total_WT=0, total_TT=0;
    float avg_WT, avg_TT;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for (i=0;i<n;i++) {
        printf("P%d Arrival: ", i+1); scanf("%d",&AT[i]);
        printf("P%d Burst: ", i+1); scanf("%d",&BT[i]);
        printf("P%d Priority: ", i+1); scanf("%d",&PR[i]);
        P[i]=i+1;
    }

    for(i=0;i<n;i++){
        pos=i;
        for(j=i+1;j<n;j++){
            if(PR[j]<PR[pos])
                pos=j;
        }
        temp=PR[i]; PR[i]=PR[pos]; PR[pos]=temp;
        temp=BT[i]; BT[i]=BT[pos]; BT[pos]=temp;
        temp=AT[i]; AT[i]=AT[pos]; AT[pos]=temp;
        temp=P[i]; P[i]=P[pos]; P[pos]=temp;
    }

    WT[0]=0;
    for(i=1;i<n;i++){
        WT[i]=0;
        for(j=0;j<i;j++)
            WT[i]+=BT[j];
        total_WT+=WT[i];
    }

    printf("\nProcess\tArrival\tBurst\tPriority\tWaiting\tTurnaround\n");
    printf("-----------------------------------------------------\n");
    for(i=0;i<n;i++){
        TT[i]=BT[i]+WT[i];
        total_TT+=TT[i];
        printf("P%d\t%d\t%d\t%d\t\t%d\t%d\n",P[i],AT[i],BT[i],PR[i],WT[i],TT[i]);
    }

    avg_WT=(float)total_WT/n;
    avg_TT=(float)total_TT/n;
    printf("\nAverage Waiting Time=%.2f",avg_WT);
    printf("\nAverage Turnaround Time=%.2f",avg_TT);
    return 0;
}

5. Round Robin Scheduling


#include <stdio.h>

int main() {
    int n, tq, BT[10], RT[10], WT[10], TT[10];
    int t=0, rem, i;
    float avg_WT=0, avg_TT=0;

    printf("Enter number of processes: ");
    scanf("%d",&n);
    printf("Enter time quantum: ");
    scanf("%d",&tq);

    rem=n;
    for(i=0;i<n;i++){
        printf("P%d Burst: ", i+1);
        scanf("%d",&BT[i]);
        RT[i]=BT[i];
    }

    while(rem!=0){
        for(i=0;i<n;i++){
            if(RT[i]>0){
                if(RT[i]>tq){
                    t+=tq;
                    RT[i]-=tq;
                } else {
                    t+=RT[i];
                    WT[i]=t-BT[i];
                    TT[i]=t;
                    avg_WT+=WT[i];
                    avg_TT+=TT[i];
                    RT[i]=0;
                    rem--;
                }
            }
        }
    }

    printf("\nProcess\tBurst\tWaiting\tTurnaround\n");
    printf("-----------------------------------\n");
    for(i=0;i<n;i++){
        printf("P%d\t%d\t%d\t%d\n",i+1,BT[i],WT[i],TT[i]);
    }

    printf("\nAverage Waiting Time: %.2f\n", avg_WT/n);
    printf("Average Turnaround Time: %.2f\n", avg_TT/n);
    return 0;
}
